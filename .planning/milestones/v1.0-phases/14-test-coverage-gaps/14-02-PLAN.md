  <behavior>
    - First call with a new ts returns false (not a duplicate)
    - Second call with same ts within TTL returns true (duplicate)
    - Call with expired ts (now + 31_000) returns false (TTL expired, re-accepted)
    - When map has MAX_SEEN_TS entries and new ts added, size stays at MAX_SEEN_TS (oldest evicted)
  </behavior>
  <action>
    Add import for `isDuplicateTs` and `MAX_SEEN_TS` from `'../slack-client.ts'` to the existing import line.

    Add a new `describe('isDuplicateTs', () => {` block at the end of the file with these tests:

    ```typescript
    describe('isDuplicateTs', () => {
      it('returns false for a ts seen for the first time', () => {
        const seenTs = new Map<string, number>()
        expect(isDuplicateTs(seenTs, '123.456', Date.now())).toBe(false)
      })

      it('returns true for a duplicate ts within TTL window', () => {
        const seenTs = new Map<string, number>()
        const now = Date.now()
        isDuplicateTs(seenTs, '123.456', now)
        expect(isDuplicateTs(seenTs, '123.456', now + 1000)).toBe(true)
      })

      it('returns false for an expired ts (re-accepted after TTL)', () => {
        const seenTs = new Map<string, number>()
        const now = Date.now()
        isDuplicateTs(seenTs, '123.456', now)
        // Simulate 31 seconds later — TTL is 30s
        expect(isDuplicateTs(seenTs, '123.456', now + 31_000)).toBe(false)
      })

      it('evicts oldest entry when seenTs reaches MAX_SEEN_TS cap', () => {
        const seenTs = new Map<string, number>()
        const now = Date.now()
        // Fill map to capacity
        for (let i = 0; i < MAX_SEEN_TS; i++) {
          isDuplicateTs(seenTs, `ts.${i}`, now)
        }
        expect(seenTs.size).toBe(MAX_SEEN_TS)
        // Adding one more should evict the oldest (ts.0) to stay at cap
        isDuplicateTs(seenTs, 'ts.new', now)
        expect(seenTs.size).toBe(MAX_SEEN_TS)
        expect(seenTs.has('ts.0')).toBe(false)
        expect(seenTs.has('ts.new')).toBe(true)
      })
    })
    ```

    Note: The cap test inserts MAX_SEEN_TS (10,000) entries — this is acceptable for a unit test since Maps are fast. The test runs in ~5ms.
